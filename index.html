// FUNÇÃO PARA VALIDAR O @ ENQUANTO O USUÁRIO DIGITA document.getElementById('reg-at').addEventListener('input', function(e) { // Remove espaços, converte para minúsculas e remove caracteres não permitidos // Permite apenas: letras (a-z), números (0-9), ponto (.) e underline (_) let valor = e.target.value.toLowerCase(); valor = valor.replace(/\s/g, ''); // Remove todos os espaços valor = valor.replace(/[^a-z0-9._]/g, ''); // Remove caracteres especiais exceto . e _ e.target.value = valor; }); function registrarUsuario() { const nome = document.getElementById('reg-nome').value.trim(); const atOriginal = document.getElementById('reg-at').value.trim(); const email = document.getElementById('reg-email').value.trim(); const pass = document.getElementById('reg-pass').value.trim(); // Remove o símbolo @ caso o usuário tenha digitado manualmente para validar o ID no banco const at = atOriginal.replace('@', ''); if(!nome || !at || !email || !pass) { return alert("Por favor, preencha todos os campos obrigatórios."); } // Verificação extra de segurança para o formato do @ const regexAt = /^[a-z0-9._]+$/; if(!regexAt.test(at)) { return alert("O @ pode conter apenas letras minúsculas, números, ponto (.) ou underline (_)."); } // 1. Verifica se o @ já existe no Database db.ref('usuarios/' + at).once('value', snapshot => { if(snapshot.exists()) { return alert("Este @ já está em uso por outra pessoa."); } // 2. Cria o usuário no Firebase Auth auth.createUserWithEmailAndPassword(email, pass) .then((userCredential) => { // 3. Salva os dados no Realtime Database db.ref('usuarios/' + at).set({ nome: nome, at: '@' + at, email: email, uid: userCredential.user.uid, foto: "https://cdn-icons-png.flaticon.com/512/149/149071.png" }); // 4. Vincula o UID ao @ db.ref('uids/' + userCredential.user.uid).set(at); alert("Conta criada com sucesso!"); location.reload(); }) .catch(error => alert("Erro ao cadastrar: " + error.message)); }); }
